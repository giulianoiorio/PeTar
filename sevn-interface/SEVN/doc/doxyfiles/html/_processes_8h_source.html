231  double hurley_final_mass(const double Ebindf, const double toll=1e-3, const double maxit=100){
232 
233  //Check
234  if (!isinisialised)
235  svlog.critical("CE process has not been initialised",__FILE__,__LINE__,sevnstd::ce_error());
236 
237  double XX = 1 + hurley_rm_x(primary->get_Z());
238 
239 
240  double MC22=secondary->getp(MHE::ID);
241  double MC1=M_core_primary;
242  double M1=primary->getp(Mass::ID);
243  double M2=secondary->getp(Mass::ID);
244  double DELY, DERI;
245  int Niter = 0;
246 
247  double Eratio = Ebindf/Ebind_ini;
248 
249  double CONST = (pow(M1+M2,XX))*(M1-MC1+M2-MC22)*Eratio;
250 
251  double MF=std::max(MC1+MC22, (M1+M2)*pow((Eratio),(1/XX)));
252  DELY = 10*toll*MF;
253 
254 
255  while(std::abs(DELY/MF)>toll && Niter<maxit){
256 
257  DELY = pow(MF,XX)*(MF-MC1-MC22) - CONST;
258 
259  DERI = pow(MF,(XX-1))*((1+XX)*MF - XX*(MC1+MC22));
260 
261  MF -= DELY/DERI;
262 
263  Niter++;
264 
265 
266 
267  }
268 
269 
270 
271  return MF;
272 
273  }
