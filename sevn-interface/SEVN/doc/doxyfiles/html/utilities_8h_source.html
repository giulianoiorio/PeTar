220  inline double kepler(const double &ecc, const double &m, const double tol = 1e-6, const int maxit = 50) { // it solves the Kepler's equation giving the mean anomaly and the eccentricity
221 
222  //We use a Generalize Newton-Raphson method that converge cubically (instead of quadratically for the classical Newton Raphson)
223  //Given the Kepler Equation we can have problem if f1 is 0, if f2 is 0 or if f1*f1-2*f*f2 is <=0
224  //f1=1-ecc*cos(E) but ecc<1 always so f1<1 as well. When f2 is 0 or f1*f1-2*f*f2 is <=0 we revert to a simple NR for that step
225 
226  int k = 1; // to check max iterations
227  double em = m; //em = eccentric anomaly (output), m = mean anomaly (input)
228  double sinem = sin(em);
229 
230  //If eccentricity is 0 we have M=E, i.e. the Eccentricity anomaly is equal to the Mean anomaly
231  if (ecc==0)
232  return m;
233 
234  double f = em - ecc*sinem - m;
235  double f1 = 1.0 - ecc*cos(em);
236  double f2 = ecc*sinem;
237  double fsquare=f1*f1 - 2*f*f2;
238  auto check_problem = [&f2,&fsquare](){ return f2==0 or fsquare<=0;}; //Lmabda expression to check if I have to pass to a NR
239 
240 
241  if (check_problem()) //If the second derivatie is 0 make a step using the first order Newthon Methond
242  em = em - f/f1;
243  else
244  em = em + (-f1 + sqrt(fsquare))/f2;
245 
246  // iterations
247  while(fabs(f) > tol && k < maxit){
248 
249 
250 
251  sinem = sin(em);
252  f = em - ecc*sinem - m;
253  f1 = 1.0 - ecc*cos(em);
254  f2 = ecc*sinem;
255  fsquare=f1*f1 - 2*f*f2;
256 
257 
258  if (check_problem()) //If the second derivatie is 0 make a step using the first order Newthon Methond
259  em = em - f/f1;
260  else
261  em = em + (-f1 + sqrt(fsquare))/f2;
262 
263 
264 
265  k++;
266  }
267 
268  return em;
269 
270  }

445  inline std::string make_pfile_str(const double plife, const size_t Phase, const unsigned int min_precision=6){
446 
447  //In the following rows we estimate plife and the transform the number to a string to initialise stars.
448  //Hovewer, if plife is very close to 0 or 1 without enough number of digits we can artificially force plife to be 0 or 1
449  //Theregore we dynamically set the precision estimating the difference between plife and 0 or 1, taking the exponent of log10
450  //and using the next larger integer.

453  unsigned int digit_1= std::ceil(std::abs(std::log10(std::abs(plife-1))));

458  //Close to plife=1 is necessary to use a large number of digits to avoid to set plife=1 when trasformi to string
459  tini_ss << "%" << std::setprecision(precision) << plife * 100 << ":" << Phase; //Starting time
460

476  template <typename T> T interpolate_1D(T xp, std::vector<T>& x_interp, std::vector<T>& y_interp, bool equispaced_interval=false, bool ext_raise=false){
477 
478 
479  if (xp<x_interp[0] and ext_raise)

481  else if (xp<=x_interp[0])

483

485  throw sevnstd::sevnerr("Error in interpolate_1D in utility.h: xp is out of boundary.");
486  else if (xp>=x_interp.back())

488 
489  size_t pos;
