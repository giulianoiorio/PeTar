129  inline unsigned int get_id_more_evolved_star() const{
130  //If one of the star is empty return the other one
131  if (star[0]->isempty and !star[1]->isempty)
132  return 1;
133  else if (star[1]->isempty)
134  return 0;
135  //If both are remnant the older is the more compact following the order HeWD,COWD,ONEWD,NS,BH
136  else if (star[0]->isremnant and star[1]->isremnant and star[0]->getp(RemnantType::ID)!=star[1]->getp(RemnantType::ID))
137  return star[0]->getp(RemnantType::ID)>star[1]->getp(RemnantType::ID) ? 0 : 1;
138  //If both are remnant of the same type, the more evolved is the one with the largest mass
139  else if (star[0]->isremnant and star[1]->isremnant)
140  return star[0]->getp(Mass::ID)>star[1]->getp(Mass::ID) ? 0 : 1;
141  //In the stars have the same phase (not remnant handled before), return the more evolved is the one with the larger plife
142  else if (star[0]->getp(Phase::ID)==star[1]->getp(Phase::ID))
143  return star[0]->plife()>=star[1]->plife() ? 0 : 1;
144  //Finally, if the stars are not empty, not remnant and don't have the same phase
145  //the more evolved is the one with the largest phase.
146  else
147  return star[0]->getp(Phase::ID)>=star[1]->getp(Phase::ID) ? 0 : 1;
148  }

666  void limit_and_correct_mass_transfer_for_donor_from_binary(Star *donor, Star *accretor, std::size_t mass_limiting_property_id){
667 
669  //This function is called inside check_nakedHe_or_nakedCO_after_binary_evolution() that is
670  //called after the update of the single stellar property and before the binary property.
671  //Therefore, the single stellar property are corrected directly using update_from_binary, while
672  //the binary property are corrected modifying the values in the VB tables of the processes using
673  //the functions modify_SemimajorDV_by_a_factor and modify_EccentricityDV_by_a_factor.
675 
676  double DM_RL_donor=0, DM_RL_accretor=0;
677 
678 
679  svlog.pdebug("Inside limit and correct mass transfer","Mdonor",donor->getp(Mass::ID),"Maccretor",accretor->getp(Mass::ID),__FILE__,__LINE__);
680 
682  //Consider all the process, if we enter in this part of the code for sure Mtot<MHe, therefore for sure DM for the donor is negative/
683  //and it is positive for the accretor. Cycle over all the processes to get the total DM variation
684  for (auto &proc : process){
685  DM_RL_donor += proc->get_var(donor->get_ID(), Mass::ID);
686  DM_RL_accretor += proc->get_var(accretor->get_ID(), Mass::ID);
687  }
688 
689  if (DM_RL_donor==0)
690  return;
691 
693  //This value will be used to rescale the accreted mass once the donated mass has been accreted.
694  double f_accreted = std::abs(DM_RL_accretor/DM_RL_donor);
695 
697  //The effective mass donated is the entire envelope (before the processes evolution) is Mtot (before evolution) = DM + Mtot(now) - MHE (now)
698  double DM_RL_donor_effective = std::abs(DM_RL_donor) + donor->getp(Mass::ID) - donor->getp(mass_limiting_property_id); //Mass that has to be
699  //The effective mass accreted is f_accreted times the effective DM
700  double DM_RL_accretor_effective = f_accreted*DM_RL_donor_effective;
701  //In order to correct the current values we add the old DM and substract the real mass lost
702  double DM_RL_donor_correction = std::abs(DM_RL_donor)-DM_RL_donor_effective; //Add the mass lost in the process evolution and subtract the effective mass loss (all are absolue values)
703  //In order to correct the current values we subctrad the old accred mass and we add the real accreted mass.
704  double DM_RL_accretor_correction = DM_RL_accretor_effective - DM_RL_accretor; //Subtract the mass accreted in the process evolution and add the effective accreted mass.
705 
706  double correction_fraction = DM_RL_accretor_effective/DM_RL_donor;
707 
709  donor->update_from_binary(Mass::ID,DM_RL_donor_correction);
710  accretor->update_from_binary(Mass::ID, DM_RL_accretor_correction);
711 
713  //The DV of the orbital properties set by the process are estimated using the old (no correct) mass transfer
714  //In this part we correct the variation of Semimajor and Eccentricity (if circularises is True the DV eccentricity is not changed)
715  //TODO The correction here assumes that DA and DE change linearly with DM but this is not really the case.
716  for (auto &proc : process){
717  //Check if the current process is causing a mass transfer, if yes correct the estimated DV by the correction fraction
718  if ( proc->is_mass_transfer_happening()){
719  proc->modify_SemimajorDV_by_a_factor(correction_fraction); //DA_new=DA_old*correction_fraction
720  if (!circularised) proc->modify_EccentricityDV_by_a_factor(correction_fraction); //DE_new=DE_old*correction_fraction
721  }
722 
723 
724  }
