315  //The BLC counter counts the evolution steps the systems has been in the BLC configuration
316  //It is increased by one if the BLC_flag is zero, while it is reset to zero when BLC_flag is false
317  unsigned int BLC_counter=0;
318 
319  for(;;) {
320 
321  try {
322 
324  binary.evolve();

326  BLC_case1 = binary.getstar(0)->amiCompact() and !binary.getstar(1)->amiCompact();
327  BLC_case2 = binary.getstar(1)->amiCompact() and !binary.getstar(0)->amiCompact();
328 
330  BLC_flag = (BLC_case1 or BLC_case2) and !binary.broken;
331 
332  //If BLC_flag is true, update the BLC counter and record state (if allowed by other flags)
333  if (BLC_flag){
334  BLC_counter++;
335  //Record only if it is time (or print only end and this is the start of the BLC conditions BLC_counter=) and if _record state is true
336  if ( ( (binary.isoutputtime() or binary.printall()) or (binary.print_only_end and BLC_counter==1) ) and _record_state ) {
337  binary.recordstate_w_timeupdate();
338  }
339  }
340  //If BLC_flag is false, but BLC_counter>0 we are at the end of a BLC phase, so record state (if allowed by other flags) ans reset the BLC counter
341  //FInally break, in fact if this system was in a BLC phase and not it is not, there is no way it will enter again in a BLC phase
342  else if(BLC_counter>0){
343  //Record only if it is time (or print only end and this is the start of the BLC conditions BLC_counter=) and if _record state is true
344  if ( ( (binary.isoutputtime() or binary.printall()) or binary.print_only_end) and _record_state ) {
345  binary.recordstate_w_timeupdate();
346  }
347  //reset the BLC counter

350  }

355  }
